      const wonIds = new Set();
      data.outreach.forEach(o => {

          wonIds.add(o['Company ID']);
        }
      });
      // Also check if company is in accounts sheet (indicates won)

        // Accounts don't have Company ID in their sheet, match by name

        if (companyName) {
          // Find matching prospect/active by name and mark as won
          [...data.prospects, ...data.active].forEach(p => {
            if (p['Company Name'] === companyName) {
              wonIds.add(p['Company ID']);
            }
          });
        }
      });

    function renderAccounts(search = '') {

      const accountSalesMap = {};

      // Aggregate sales by company name
      data.sales.forEach(sale => {
        const companyName = sale['Company Name'];
        if (!companyName) return;
        if (!accountSalesMap[companyName]) {
          accountSalesMap[companyName] = {
            totalRevenue: 0,
            transactionCount: 0,
            sales: []
          };
        }
        accountSalesMap[companyName].totalRevenue += parseFloat(sale['Payment Amount']) || 0;
        accountSalesMap[companyName].transactionCount++;
        accountSalesMap[companyName].sales.push(sale);

      let accountsWithSales = data.accounts.map(account => {
        const companyName = account['Company Name'];
        const salesData = accountSalesMap[companyName] || {
          totalRevenue: 0,
          transactionCount: 0,
          sales: []
        };

        return {
          ...account,
          totalRevenue: salesData.totalRevenue,
          transactionCount: salesData.transactionCount,
          sales: salesData.sales
